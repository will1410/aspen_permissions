// ==UserScript==
// @name         Aspen Discovery Admin Tree Extractor (Group Indentation)
// @namespace    http://tampermonkey.net
// @version      9.0
// @description  Traverses Aspen Admin deeply with custom double-space padding and deeper indentation after p.h2 group elements.
// @author       Your Name
// @match        *://*/Admin/*
// @grant        none
// ==/UserScript==

(function () {
	'use strict';

	// 1. Floating Action UI Container Setup
	const container = document.createElement('div');
  container.id = 'open_copy';
	container.style.position = 'fixed';
	container.style.bottom = '20px';
	container.style.right = '20px';
	container.style.zIndex = '99999';
	container.style.display = 'flex';
	container.style.gap = '10px';
	document.body.appendChild(container);

	// 2. Action Trigger 1: Forced Unroll/Expansion
	const openBtn = document.createElement('button');
	openBtn.innerText = '🔓 1. Open Panels';
	openBtn.style.padding = '12px 18px';
	openBtn.style.backgroundColor = '#ffc107';
	openBtn.style.color = '#000';
	openBtn.style.border = 'none';
	openBtn.style.borderRadius = '5px';
	openBtn.style.cursor = 'pointer';
	openBtn.style.fontWeight = 'bold';
	container.appendChild(openBtn);

	// 3. Action Trigger 2: Structural Copy Target
	const copyBtn = document.createElement('button');
	copyBtn.innerText = '📋 2. Copy Labels';
	copyBtn.style.padding = '12px 18px';
	copyBtn.style.backgroundColor = '#0056b3';
	copyBtn.style.color = '#fff';
	copyBtn.style.border = 'none';
	copyBtn.style.borderRadius = '5px';
	copyBtn.style.cursor = 'pointer';
	copyBtn.style.fontWeight = 'bold';
	container.appendChild(copyBtn);

	// --- LOGIC BLOCK 1: FORCED STRUCTURAL EXPANSION ---
	openBtn.addEventListener('click', function () {
		const hiddenStructuralSelectors = [
			'.panel-collapse', '.collapse', '.accordion-body',
			'.hidden', '.hide', '[aria-hidden="true"]'
		];

		document.querySelectorAll(hiddenStructuralSelectors.join(', ')).forEach(el => {
			el.classList.remove('collapse', 'hidden', 'hide', 'collapsed');
			el.classList.add('in', 'show');
			el.style.setProperty('display', 'block', 'important');
			el.style.setProperty('height', 'auto', 'important');
			el.style.setProperty('visibility', 'visible', 'important');
			if (el.hasAttribute('aria-hidden')) el.setAttribute('aria-hidden', 'false');
		});

		document.querySelectorAll('[aria-expanded="false"], .collapsed').forEach(toggle => {
			toggle.setAttribute('aria-expanded', 'true');
			toggle.classList.remove('collapsed');
		});

		showToastNotice('🔓 All configuration panels forced open!');
	});

	// --- LOGIC BLOCK 2: DEEP LOGICAL EXTRACTION TREE ---
	copyBtn.addEventListener('click', function () {
		try {
			const allElements = Array.from(document.querySelectorAll('*'));

			// Boundary Elements Setup
			let startElement = allElements.find(el => {
				if (el.closest('#side-bar') || el.closest('#sidebar') || el.closest('.sidebar')) return false;
				const txt = (el.placeholder || el.value || el.innerText || '').toLowerCase();
				return txt.includes('search for a property');
			});

			let endElement = allElements.find(el => {
				if (el.closest('#side-bar') || el.closest('#sidebar') || el.closest('.sidebar')) return false;
				const val = (el.value || el.title || el.innerText || '').toLowerCase();
				return val.includes('save changes and return') || val.includes('save changes');
			});

			let rawTreeLines = [];

			// Continuous fall-through structural mapping function
			function exploreNode(node, depth = 0) {
				if (!node || node.tagName === 'SCRIPT' || node.tagName === 'STYLE' || node === openBtn || node === copyBtn) return;

				// Instantly scrub and drop sidebar items
				if (node.id === 'side-bar' || node.id === 'sidebar' || node.classList?.contains('sidebar') || node.closest('#side-bar') || node.closest('#sidebar') || node.closest('.sidebar')) {
					return;
				}

				const indent = '  '.repeat(depth);
				let nextDepth = depth;

				// 1. Identify Panel containers
				const isPanel = node.classList?.contains('panel') ||
					node.classList?.contains('card') ||
					node.classList?.contains('box') ||
					node.classList?.contains('accordion-group') ||
					node.tagName === 'FIELDSET';

				if (isPanel) {
					const heading = node.querySelector('.panel-heading, .panel-title, .card-header, .accordion-toggle, legend, h1, h2, h3, h4');
					const labelText = heading ? heading.innerText.trim().replace(/\n+/g, ' ') : '[Panel Section]';

					rawTreeLines.push({ element: node, text: `${indent}${labelText}` });
					nextDepth = depth + 1; // Increment indentation depth for nested content
				}

				// 2. Identify Group Headings marked strictly by p.h2
				const isParagraphHeading = node.tagName === 'P' && node.classList?.contains('h2');

				if (isParagraphHeading) {
					const headingText = node.innerText.trim().replace(/\n+/g, ' ');
					if (headingText) {
						// Insert double space (two blank line strings) directly before the header line
						rawTreeLines.push({ element: node, text: "" });

						// Push the group header text without the 📁 icon
						rawTreeLines.push({ element: node, text: `${indent}${headingText}` });

						// Push the indentation out by 1 extra level for all downstream children inside this p.h2 section
						nextDepth = nextDepth + 1;
					}
				}

				// 3. Identify and map Form Elements
				const isInput = ['INPUT', 'SELECT', 'TEXTAREA'].includes(node.tagName);
				if (isInput && node.type !== 'hidden') {
					let associatedLabel = '';

					if (node.id) {
						const labelEl = document.querySelector(`label[for="${node.id}"]`);
						if (labelEl) associatedLabel = labelEl.innerText.trim();
					}
					if (!associatedLabel) {
						associatedLabel = node.closest('.form-group')?.querySelector('label')?.innerText.trim() || '';
					}

					if (associatedLabel) {
						const cleanLabel = associatedLabel.replace(/\n+/g, ' ').replace(/\s+/g, ' ');
						rawTreeLines.push({ element: node, text: `${indent}${cleanLabel}` });
					}
				}

				// Recursive down-crawl through downstream children
				Array.from(node.children).forEach(child => {
					exploreNode(child, nextDepth);
				});
			}

			// Canvas wrapper detection
			const rootContainer = document.getElementById('main-content-with-sidebar') || document.querySelector('form') || document.body;
			exploreNode(rootContainer, 0);

			let textToCopy = "";

			// Trim output content down to target boundaries if matching elements exist
			if (startElement && endElement) {
				let finalTree = rawTreeLines.filter(item => {
					const positionFromStart = startElement.compareDocumentPosition(item.element);
					const positionFromEnd = item.element.compareDocumentPosition(endElement);

					const isAfterStart = (positionFromStart & Node.DOCUMENT_POSITION_FOLLOWING) || startElement.contains(item.element);
					const isBeforeEnd = (positionFromEnd & Node.DOCUMENT_POSITION_FOLLOWING) || endElement.contains(item.element);

					return isAfterStart && isBeforeEnd;
				}).map(item => item.text);

				textToCopy = finalTree.join('\n');
			}

			// Safe fallback logic
			if (!textToCopy) {
				textToCopy = rawTreeLines.map(i => i.text).join('\n');
			}

			if (!textToCopy.trim()) {
				alert('Parsing Error: The configuration fields container area came up blank.');
				return;
			}

			// Clipboard Action Pipeline
			const textArea = document.createElement("textarea");
			textArea.value = textToCopy;
			textArea.style.position = "fixed";
			textArea.style.top = "0";
			textArea.style.left = "0";
			textArea.style.width = "2em";
			textArea.style.height = "2em";
			textArea.style.padding = "0";
			textArea.style.border = "none";
			textArea.style.outline = "none";
			textArea.style.background = "transparent";

			document.body.appendChild(textArea);
			textArea.focus();
			textArea.select();

			const successful = document.execCommand('copy');
			document.body.removeChild(textArea);

			if (successful) {
				showToastNotice(`✅ Copied ${textToCopy.split('\n').length} items!`);
			} else {
				throw new Error('Clipboard permissions error');
			}

		} catch (globalError) {
			alert('Execution Trace Exception:\n' + globalError.message);
			console.error(globalError);
		}
	});
	// Helper: Toast Status Alert Rendering
	function showToastNotice(msg) { const toast = document.createElement('div'); toast.innerText = msg; toast.style.position = 'fixed'; toast.style.bottom = '80px'; toast.style.right = '20px'; toast.style.backgroundColor = '#333'; toast.style.color = '#fff'; toast.style.padding = '8px 16px'; toast.style.borderRadius = '4px'; toast.style.zIndex = '100000'; document.body.appendChild(toast); setTimeout(() => toast.remove(), 3000); }
})();
